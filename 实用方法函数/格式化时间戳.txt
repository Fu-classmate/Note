//格式化时间戳   Fri Jan 29 2021 17:26:33 GMT+0800 (中国标准时间)
formatDate(date) {
        let d = new Date(date)
        let month = '' + (d.getMonth() + 1)
        let day = '' + d.getDate()
        let year = d.getFullYear()

        if (month.length < 2) month = '0' + month;
        if (day.length < 2) day = '0' + day;

        return [year, month, day].join('-')
}
